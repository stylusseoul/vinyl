// ============================================================================== // 🔥 신청곡 & 어드민 기능 (단일 업데이트 & 충돌 방지 락 적용) // ============================================================================== const SHEET_ID = '1XnYwp6cxnesB0vaFZ955QNpLXnWMsfrob7Ga2-UtvMw'; const REQUESTS_SHEET = 'Requests'; /** * OPTIONS 핸들러 (CORS 대응) */ function doOptions(e) { return ContentService.createTextOutput("OK").setMimeType(ContentService.MimeType.TEXT); } /** * POST 핸들러 */ function doPost(e) { // ★ 동시에 요청이 들어올 경우를 대비해 10초간 시트 접근에 락을 겁니다. const lock = LockService.getScriptLock(); lock.waitLock(10000); try { const postData = e.postData ? e.postData.contents : ''; if (!postData) return createJsonResponse_({ result: 'error', message: 'No data' }); let parsedData = null; let isBase64 = false; // 1단계: JSON vs Base64 식별 try { parsedData = JSON.parse(postData); } catch (parseError) { isBase64 = true; } // [분기 1] 어드민 단일 업데이트 (락 적용) if (!isBase64 && parsedData && parsedData.timestamp) { const targetTimestamp = String(parsedData.timestamp).trim(); const newStatus = parsedData.status || ''; // active가 아닌 ID로 정확히 시트 타겟팅 const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(REQUESTS_SHEET); const aColumnRange = sheet.getRange("A:A"); const textFinder = aColumnRange.createTextFinder(targetTimestamp).matchEntireCell(false); const searchResult = textFinder.findNext(); if (searchResult) { const foundRow = searchResult.getRow(); sheet.getRange(foundRow, 8).setValue(newStatus); SpreadsheetApp.flush(); // 즉시 기록 강제 return createJsonResponse_({ result: 'success' }); } else { return createJsonResponse_({ result: 'error', message: 'Not found' }); } } // [분기 2] 웹사이트에서 새 신청곡이 들어온 경우 (Base64) if (isBase64) { const decodedBytes = Utilities.base64Decode(postData); const decodedString = Utilities.newBlob(decodedBytes).getDataAsString('UTF-8'); const data = JSON.parse(decodedString); const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(REQUESTS_SHEET); const configRange = sheet.getRange('A1:B1').getValues()[0]; if (configRange[1] !== 'ON') { return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } const kstFormat = Utilities.formatDate(new Date(), "Asia/Seoul", "yyyy.MM.dd HH:mm"); const aColumn = sheet.getRange("A:A").getValues(); let emptyRow = 1; for (let i = 1; i < aColumn.length; i++) { if (aColumn[i][0] === "") { emptyRow = i + 1; break; } } if (emptyRow === 1) emptyRow = sheet.getLastRow() + 1; sheet.getRange(emptyRow, 1, 1, 7).setValues([[ kstFormat, data.artist || '', data.album || '', data.track || '', data.requester || '익명', data.memo || '', data.cover || '' ]]); SpreadsheetApp.flush(); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } return createJsonResponse_({ result: 'error', message: 'Unknown format' }); } catch (error) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(REQUESTS_SHEET); sheet.appendRow([new Date(), "ERROR", error.toString(), "RawData: " + e.postData.contents]); } catch (e2) {} return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } finally { // ★ 에러가 나든 성공하든 무조건 락 해제 lock.releaseLock(); } } /** * GET 핸들러 */ function doGet(e) { try { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(REQUESTS_SHEET); const configRange = sheet.getRange('A1:B1').getValues()[0]; return createJsonResponse_({ enabled: configRange[1] === 'ON', status: configRange[1] || 'OFF' }); } catch (error) { return createJsonResponse_({ enabled: false, status: 'ERROR' }); } } function createJsonResponse_(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
